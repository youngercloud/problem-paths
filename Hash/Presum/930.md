本题和 560 一模一样， 只不过把所有数字换成了 0 或者 1 而已， 其余部分没变化

```
class Solution {
public:
    int numSubarraysWithSum(vector<int>& nums, int goal) {
        int m = nums.size();
        unordered_map<int, int> Map;
        int cur = 0;
        int ret = 0;
        for (int i = 0; i < m; i++) {
            cur += nums[i]; 
            if (cur == goal) ret++;
            ret += Map[cur - goal];
            Map[cur]++;
        }
        return ret;
    }
};
```
