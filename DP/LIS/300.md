1. DP
```
        int m = nums.size();
        vector<int> dp(m + 1, 1);
        int res = 1;
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= i; j++) {
                if (nums[i - 1] > nums[j - 1]) {
                    dp[i] = max(dp[i], dp[j] + 1);    
                    res = max(res, dp[i]);
                }
            }
        }
        return res;
```
