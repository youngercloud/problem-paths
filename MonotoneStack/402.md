class Solution {
public:
    string removeKdigits(string num, int k) {
        string s = "";
        for (auto c : num) {
            while (!s.empty() && c < s.back() && k) {
                s.pop_back();
                k--;
            }
            s.push_back(c);
        }
        
        while (!s.empty() && k > 0) {
            s.pop_back();
            k--;
        }
        
        while (s[0] == '0') s.erase(s.begin());
        return s == "" ? "0" : s;
    }
};
