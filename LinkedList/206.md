    ListNode* reverseList(ListNode* head) {
        ListNode* prev = NULL;              // 反转后头部应该指向 NULL
        ListNode* cur = head;               // 从头部开始反转
        
        while (cur != NULL) {
            ListNode* temp = cur->next;     // 保留下一个要去的节点
            cur->next = prev;               // 反转
            
            prev = cur;                     // 向前走
            cur = temp;
        }
        return prev;
    }
